ACTOR ToiletShootable //3d actor
{
	//$Category Props (Wolf3D)/Bath
	//$Title Toilet (shootable)
	//$Color 3
	DistanceCheck "scenelod"
	Health 1
	Radius 10
	Height 42
	Mass 999999
	+DONTGIB
	+NOBLOOD
	+NOICEDEATH
	+NOTAUTOAIMED
	+SHOOTABLE
	+SOLID
	DeathSound "toilet/death"
	States
	{
	Spawn:
		MDLA A -1
		Stop
	Death:
		MDLA A 2 A_Scream
		TOIL BC 2
	Broken:
		TOIL D 2 A_PlaySound ("toilet/broken", 0, 1.0, 1, 5)
		"####" EFG 2
		Loop
	}
}
