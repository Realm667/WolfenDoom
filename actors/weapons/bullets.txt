ACTOR Tracer : FastProjectile
{
Alpha 0.90
RenderStyle Add
Radius 2
Height 2
Speed 75
PROJECTILE
+BLOODSPLATTER
Scale 0.4
MissileType TracerTrail
MissileHeight 8
Decal BulletChip
States
	{
	Spawn:
	PUFF A 1 BRIGHT
	PUFF A 1 BRIGHT LIGHT(TRCRFIRE) A_CheckSightOrRange(192,"Whiz")
	Loop
	Whiz:
	PUFF A 1 BRIGHT LIGHT(TRCRFIRE) A_PlaySound("whiz",1.0,0,ATTN_NONE)
	SpawnNoWhiz:
	PUFF A 1 BRIGHT
	Loop
	Death:
	Crash:
	PUFF A 2 BRIGHT LIGHT(TRCRFIRE) A_SpawnDebris("TracerSpark")
	PUFF B 1 BRIGHT A_SetTranslucent(0.9,0)
	PUFF B 1 BRIGHT A_PlaySound("ricochet")
	PUFF CD 2 BRIGHT
	Stop
	XDeath:
	TNT1 A 3 A_PlaySound("hitflesh")
	Stop
	}
}
