		Goto Ready

	Radius 16
	Scale 0.1

	+BLOODLESSIMPACT
	+RIPPER
	+WINDTHRUST

	DamageType Fire
	Decal Scorch
	Obituary "$OBPYRO"
	ProjectileKickback 20
	SpawnID 222
	States
	{
	Spawn:
		"####" A 0 NODELAY A_Jump(256, "Flame1", "Flame2", "Flame3", "Flame4")
		Stop
	Flame1:
		FLM1 A 0 
		Goto Animation
	Flame2:
		FLM2 A 0 
		Goto Animation
	Flame3:
		FLM3 A 0 
		Goto Animation
	Flame4:
		FLM4 A 0 
		Goto Animation
	Animation:
		"####" AAAAA 3 BRIGHT LIGHT(BOAFLAMW) 
			{
				A_SetScale(scalex+0.05);
				A_FadeOut(0.05);
				A_Explode(2,16);
			}
		"####" A 0 A_Jump(128, "Smoke1", "Smoke2", "Smoke3")
		Goto Death
	Smoke1:
		"####" A 0 A_SpawnItemEx("Smoke1A")
		Goto Death
	Smoke2:
		"####" A 0 A_SpawnItemEx("Smoke2A")

	}
