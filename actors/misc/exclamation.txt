ACTOR Exclamation : SwitchableDecoration
{
	//$Category Misc (Wolf3D)
	//$Title Exclamation Mark
	//$Color 1
	Radius 32
	Height 64
	Scale 1.35
	RenderStyle "Translucent"
	Alpha 1.0
	+CLIENTSIDEONLY
	+NOGRAVITY
	States
	{
	Spawn:
		//TNT1 A -1 NODELAY ACS_NamedExecuteAlways("BoA_CompassQueue",0)
	Active:
		EXCL A -1 BRIGHT
		Loop
	Inactive:
		TNT1 A -1
		Stop
	}
}

ACTOR ExclamationTouchable : Exclamation
{
	//$Category Misc (Wolf3D)
	//$Title Exclamation Mark (inactive on touching)
	//$Color 1
	States
	{
	Spawn:
		//TNT1 A -1 NODELAY ACS_NamedExecuteAlways("BoA_CompassQueue",0)
	Active:
		EXCL A 1 NODELAY BRIGHT A_CheckProximity("Inactive", "BoAPlayer", 72, 1)
		Loop
	Inactive:
		EXCL A 35 BRIGHT
		EXCL B -1 BRIGHT
		Stop
	}	
}

ACTOR InteractionIcon
{
	Height 8
	Radius 64
	Scale 1.0
	RenderStyle "Translucent"
	Alpha 1.0
	-SOLID
	+CLIENTSIDEONLY
	+FLOATBOB
	+NOBLOCKMAP
	+NOGRAVITY
	States
	{
	Spawn:
		TNT1 A 0 NoDelay A_CheckRange(128.0, "FadeAway", TRUE)
		ICNI A 1 BRIGHT
		Loop
	FadeAway:
		"####" A 10
		Goto Spawn
	}	
}
