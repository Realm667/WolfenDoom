ACTOR StarParticle 
{
	Gravity 0.125
	Radius 1
	Height 1
	RenderStyle "Add"
	Alpha 1.0
	YScale 1.0
	XScale 1.0
	+FORCEXYBILLBOARD
	+MISSILE
	+NOGRAVITY
	+THRUACTORS
	States
	{
	Spawn:
		EMB5 I 1200 BRIGHT 
		EMB5 A 0 A_Die
	Death:
		Stop
	}
}

ACTOR StarParticle_Spawner : SwitchableDecoration
{
	//$Category Special Effects (Wolf3D)
	//$Title Startdust Spawner (horizontal)
	//$Color 12
	-SOLID
	+CLIENTSIDEONLY
	+NOCLIP
	+NOGRAVITY
	+NOINTERACTION
	RenderStyle "Add"
	Alpha 1.0
	States
	{
	Spawn:
		TNT1 A 0 NODELAY A_JumpIf(CallACS("DustSwitch")==0,"EndSpawn")
	Active:
		TNT1 A 1
		TNT1 A 5 A_SpawnItemEx ("StarParticle", random(0,128), 0, random(0,128), 0, random(2,8), 0, 0, SXF_CLIENTSIDE, 64)
		Loop
	Inactive:
		TNT1 A 10
		Loop
	EndSpawn:
		TNT1 A 1
		Stop
	}
}
