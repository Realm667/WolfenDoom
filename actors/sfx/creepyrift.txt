//This actor is based over Maelstrom's Phantom attacks
ACTOR CreepyGhost : Base
{
	Radius 16
	Height 32
	Health 1
	Mass 500
	Speed 16
	Damage 1
	DamageType "Creepy"
	Projectile
	-SOLID
	+FLOAT
	+FRIENDLY
	+LOOKALLAROUND
	+NOBLOCKMONST
	+NOBLOOD
	+NOGRAVITY
	RenderStyle "Add"
	Alpha 0.5
	Base.Shadow 0  //needed for shadows
	States
		{
		Spawn:
			SPI1 A 0 A_StartSound("creepy/born")
		See:
			"####" AABB 2 A_FastChase
			Loop
		Melee:
			"####" A 4 A_FaceTarget
			"####" B 2 A_Die
			Stop
		Death:
			"####" E 0 A_Explode(100,Radius+16)
			"####" E 0 A_StartSound("creepy/dies")
		DeathLoop:
			"####" FGHIJ 2
			Stop
		}
}
