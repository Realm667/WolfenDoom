ACTOR Flare1 : SwitchableDecoration
{
	//$Category Special Effects (Wolf3D)
	//$Title Flare (white)
	//$Color 12
	Height 16
	Radius 16
	+NOINTERACTION
	+NOBLOCKMAP
	+NOGRAVITY
	+NOCLIP
	Scale 0.5
	RenderStyle "Add"
	States
	{
	Spawn:
	Active:
		FLAR A -1
		Loop
	Inactive:
		TNT1 A -1
		Loop
	}
}
